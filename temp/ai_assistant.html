function calculateDiscount(price, isVIP) {


  let discount = 0;


  if (price > 1000) {


    discount = price * 0.1;


  } else if (price > 500) {


    discount = price * 0.05;


  }


  if (isVIP) {


    discount += price * 0.05;


  }


  return price - discount;


}

สนทนากับ AI ผู้ช่วย

สวัสดีครับ! มีอะไรให้ช่วยเกี่ยวกับการเขียนโค้ดไหมครับ?

ช่วยอธิบายว่าทำไมเราถึงใช้ if แทนที่จะใช้ switch ในฟังก์ชัน calculateDiscount ได้ไหม?

ในฟังก์ชัน calculateDiscount เราใช้ if เพราะ:

  1. เงื่อนไขเป็นการเปรียบเทียบช่วงของค่า (price > 1000, price > 500) ซึ่ง switch ไม่เหมาะกับการเปรียบเทียบแบบนี้
  2. มีการตรวจสอบเงื่อนไขหลายอย่างที่ไม่เกี่ยวข้องกัน (ราคาและสถานะ VIP)
  3. switch จะเหมาะกับการเปรียบเทียบค่าที่เท่ากัน (===) เท่านั้น

ถ้าต้องการใช้ switch ต้องปรับโครงสร้างโค้ดใหม่ เช่น:

function calculateDiscountWithSwitch(price, isVIP) {
  let discount = 0;
  
  // ต้องแปลงเงื่อนไขให้เป็นค่าที่แน่นอน
  const priceRange = price > 1000 ? 'high' : 
                     price > 500 ? 'medium' : 'low';
                     
  switch(priceRange) {
    case 'high':
      discount = price * 0.1;
      break;
    case 'medium':
      discount = price * 0.05;
      break;
    case 'low':
      discount = 0;
      break;
  }
  
  if (isVIP) {
    discount += price * 0.05;
  }
  
  return price - discount;
}
